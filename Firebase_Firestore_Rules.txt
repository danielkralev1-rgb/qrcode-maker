// Firestore security rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAuthed() { return request.auth != null; }
    match /items/{sku} {
      allow read, write: if isAuthed();
    }
    match /transactions/{doc} {
      allow read, write: if isAuthed();
    }
  }
}
